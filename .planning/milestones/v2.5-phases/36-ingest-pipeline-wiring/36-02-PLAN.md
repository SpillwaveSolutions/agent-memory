5. **Add GetDedupStatus RPC handler** (crates/memory-service/src/ingest.rs)

Add the proto imports for `GetDedupStatusRequest` and `GetDedupStatusResponse` to the import block at the top.

Add the RPC implementation in the `impl MemoryService for MemoryServiceImpl` block:
```rust
async fn get_dedup_status(
    &self,
    _request: Request<GetDedupStatusRequest>,
) -> Result<Response<GetDedupStatusResponse>, Status> {
    let response = if let Some(ref checker) = self.novelty_checker {
        let config = checker.config();
        let metrics = checker.metrics();
        let snapshot = metrics.snapshot();
        GetDedupStatusResponse {
            enabled: config.enabled,
            threshold: config.threshold,
            events_checked: snapshot.total_checked(),
            events_deduplicated: snapshot.total_rejected(),
            events_skipped: snapshot.total_stored() - snapshot.stored_novel,
            buffer_size: 0,  // TODO: expose buffer len if needed
            buffer_capacity: config.buffer_capacity as u32,
        }
    } else {
        GetDedupStatusResponse {
            enabled: false,
            threshold: 0.0,
            events_checked: 0,
            events_deduplicated: 0,
            events_skipped: 0,
            buffer_size: 0,
            buffer_capacity: 0,
        }
    };
    Ok(Response::new(response))
}
```

Note: The `snapshot()` method returns `NoveltyMetricsSnapshot` which has `total_checked()`, `total_rejected()`, `total_stored()`, and `stored_novel` field. Use these to populate the response.

Also import `NoveltyMetricsSnapshot` if needed (or just use the snapshot inline).

```rust
use memory_service::novelty::{NoveltyChecker, CandleEmbedderAdapter};
use memory_types::dedup::InFlightBuffer;

```

Create the checker with fail-open on CandleEmbedder initialization:

```rust
let novelty_checker = if settings.dedup.enabled {
    match memory_embeddings::CandleEmbedder::load(&memory_embeddings::ModelCache::default()) {
        Ok(embedder) => {
            let adapter = Arc::new(CandleEmbedderAdapter::new(embedder)) as Arc<dyn memory_service::novelty::EmbedderTrait>;
            let buffer = Arc::new(RwLock::new(
                InFlightBuffer::new(settings.dedup.buffer_capacity, 384)
            ));
            let checker = NoveltyChecker::with_in_flight_buffer(
                Some(adapter),
                buffer,
                settings.dedup.clone(),
            );
            info!("Dedup gate enabled (threshold: {}, buffer: {})", settings.dedup.threshold, settings.dedup.buffer_capacity);
            Some(Arc::new(checker))
        }
        Err(e) => {
            tracing::warn!("Failed to load CandleEmbedder for dedup, disabling: {e}");
            None
        }
    }
} else {
    debug!("Dedup gate disabled by config");
    None
};
```

Then pass it to the server:
```rust
let result = run_server_with_scheduler(addr, storage, scheduler, shutdown_signal, novelty_checker).await;
```
