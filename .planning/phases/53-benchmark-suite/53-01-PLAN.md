```rust
use anyhow::{bail, Result};
use serde::{Deserialize, Serialize};
use std::path::Path;

#[derive(Debug, Deserialize, Serialize)]
pub struct Fixture {
    #[serde(rename = "test")]
    pub tests: Vec<TestCase>,
}

#[derive(Debug, Deserialize, Serialize, Clone)]
pub struct TestCase {
    pub id: String,
    pub description: String,
    pub setup: Vec<String>,       // Paths to JSONL session files
    pub query: String,
    pub expected_contains: Vec<String>,  // Case-insensitive substring match
    pub max_tokens: usize,
}

impl Fixture {
    pub fn load(path: &Path) -> Result<Self> { /* validate id non-empty, query non-empty */ }
    pub fn load_dir(dir: &Path) -> Result<Vec<TestCase>> { /* read all .toml files in dir */ }
}
```
