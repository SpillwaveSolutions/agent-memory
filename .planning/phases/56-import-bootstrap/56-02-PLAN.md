**2. Create import.rs command module:**

The import command reads the backup directory structure and streams ImportChunk messages to the daemon.

```rust
//! `memory import` command -- restore memory from a backup directory.

use anyhow::{bail, Context, Result};
use std::path::{Path, PathBuf};

use memory_client::{BackupChunkType, ImportChunk, ImportResult};

use crate::cli::{GlobalArgs, ImportArgs};
use crate::client::connect_client;
```

**run function flow:**
1. Connect to daemon via `connect_client`
2. Read and validate `manifest.json` from backup dir — check `"version": "1.0"` exists. If version missing or unsupported, bail with error (fail fast).
3. Build a `Vec<ImportChunk>` by reading files in order:
   a. Read `events/*.jsonl` files sorted by filename (chronological). For each file, read lines, batch into chunks of ~100 lines per ImportChunk with `chunk_type = BackupChunkType::Events`.
   b. If NOT events_only: read `toc/segments.jsonl` (TocSegments), `toc/days.jsonl` (TocDays), `toc/weeks.jsonl` (TocWeeks), `toc/months.jsonl` (TocMonths), `toc/years.jsonl` (TocYears) — each batched into ImportChunks with appropriate BackupChunkType.
   c. If NOT events_only: read `grips.jsonl` (Grips) and `episodes.jsonl` (Episodes).
4. Set `dry_run` and `events_only` flags on every ImportChunk.
5. Convert `Vec<ImportChunk>` to stream using `tokio_stream::iter(chunks)`.
6. Call `client.import_backup(stream).await` to get `ImportResult`.
7. Report results to stderr.

**File reading helper:**
```rust
fn read_jsonl_chunks(
    path: &Path,
    chunk_type: BackupChunkType,
    dry_run: bool,
    events_only: bool,
    batch_size: usize,
) -> Result<Vec<ImportChunk>> {
    if !path.exists() {
        return Ok(Vec::new());
    }
    let content = std::fs::read_to_string(path)
        .with_context(|| format!("Failed to read {}", path.display()))?;
    let lines: Vec<&str> = content.lines().filter(|l| !l.is_empty()).collect();
    let mut chunks = Vec::new();
    for batch in lines.chunks(batch_size) {
        chunks.push(ImportChunk {
            chunk_type: chunk_type as i32,
            jsonl_data: batch.join("\n"),
            record_count: batch.len() as u32,
            dry_run,
            events_only,
        });
    }
    Ok(chunks)
}
```

**Manifest validation:**
```rust
fn validate_manifest(base: &Path) -> Result<serde_json::Value> {
    let manifest_path = base.join("manifest.json");
    let content = std::fs::read_to_string(&manifest_path)
        .context("Failed to read manifest.json — is this a valid backup directory?")?;
    let manifest: serde_json::Value = serde_json::from_str(&content)
        .context("Failed to parse manifest.json")?;
    let version = manifest.get("version").and_then(|v| v.as_str()).unwrap_or("");
    if version != "1.0" {
        bail!("Unsupported backup version: '{}' (expected '1.0')", version);
    }
    Ok(manifest)
}
```

#[tokio::test]
async fn test_import_events_round_trip() {
    let dir = TempDir::new().unwrap();
    let storage = make_storage(&dir);

    // Build a known event as JSONL
    let event = Event {
        event_id: "test-event-001".to_string(),
        timestamp_ms: 1_700_000_000_000,
        content: "hello world".to_string(),
        ..Default::default()
    };
    let jsonl = serde_json::to_string(&event).unwrap();

    let chunk = ImportChunk {
        chunk_type: BackupChunkType::Events as i32,
        jsonl_data: jsonl,
        record_count: 1,
        dry_run: false,
        events_only: false,
    };

    let req = make_stream(vec![chunk]);
    let result = import::import_backup(storage.clone(), req).await.unwrap().into_inner();

    assert_eq!(result.events_imported, 1);
    assert_eq!(result.events_skipped, 0);
    assert_eq!(result.errors, 0);

    // Verify event is retrievable from storage
    let retrieved = storage.get_event("test-event-001").expect("get_event failed");
    assert!(retrieved.is_some(), "Event should exist in storage after import");
    let retrieved_event = retrieved.unwrap();
    assert_eq!(retrieved_event.event_id, "test-event-001");
    assert_eq!(retrieved_event.content, "hello world");
}

#[tokio::test]
async fn test_import_events_idempotent() {
    let dir = TempDir::new().unwrap();
    let storage = make_storage(&dir);

    let event = Event {
        event_id: "dedup-event-001".to_string(),
        timestamp_ms: 1_700_000_000_000,
        content: "deduplicated".to_string(),
        ..Default::default()
    };
    let jsonl = serde_json::to_string(&event).unwrap();

    // Import twice
    for _ in 0..2 {
        let chunk = ImportChunk {
            chunk_type: BackupChunkType::Events as i32,
            jsonl_data: jsonl.clone(),
            record_count: 1,
            dry_run: false,
            events_only: false,
        };
        let req = make_stream(vec![chunk]);
        import::import_backup(storage.clone(), req).await.unwrap();
    }

    // Second import should have skipped the duplicate
    let chunk = ImportChunk {
        chunk_type: BackupChunkType::Events as i32,
        jsonl_data: jsonl.clone(),
        record_count: 1,
        dry_run: false,
        events_only: false,
    };
    let req = make_stream(vec![chunk]);
    let result = import::import_backup(storage.clone(), req).await.unwrap().into_inner();
    assert_eq!(result.events_skipped, 1, "Duplicate event should be skipped");
    assert_eq!(result.events_imported, 0);
}
